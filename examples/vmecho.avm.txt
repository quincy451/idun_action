entry start

start:
  push16 80
  calln reu_alloc
  store 0
  push16 0
  store 1

  setp16 prompt
  calln print

read_loop:
  calln conin
  dup
  push8 13
  eq
  jz store_char

  drop
  setp16 prefix
  calln print
  push16 0
  store 2

echo_loop:
  load 2
  load 1
  eq
  jz emit_char

  load 0
  calln reu_free
  setp16 blank
  calln printe
  calln exit

emit_char:
  load 0
  load 2
  calln reu_peek8
  calln conout
  load 2
  push8 1
  add
  store 2
  jmp echo_loop

store_char:
  store 3
  load 0
  load 1
  load 3
  calln reu_poke8
  load 1
  push8 1
  add
  store 1
  jmp read_loop

prompt:
  stringz "type> "
prefix:
  stringz "stored:"
blank:
  stringz ""
