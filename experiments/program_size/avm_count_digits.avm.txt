entry start

start:
  push16 in_name
  calln fopenr
  jz exit_now
  push16 0
  store 0

read_loop:
  calln fread8
  dup
  push16 $ffff
  eq
  jz have_byte
  drop
  load 0
  calln printie
  calln fcloser
  calln exit

have_byte:
  store 1
  load 1
  push8 48
  lt
  jz chk_hi
  jmp read_loop
chk_hi:
  load 1
  push8 57
  gt
  jz count_it
  jmp read_loop
count_it:
  load 0
  push16 1
  add
  store 0
  jmp read_loop

exit_now:
  calln exit

in_name:
  stringz "DIGITS.TXT"
