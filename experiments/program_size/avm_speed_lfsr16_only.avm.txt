entry start

start:
  setp16 start_msg
  calln printe
  push16 $ACE1
  store 0
  push16 60000
  store 1

loop:
  load 0
  push16 1
  band
  jz even
  load 0
  shr1
  push16 $B400
  bxor
  store 0
  jmp dec

even:
  load 0
  shr1
  store 0

dec:
  load 1
  push16 1
  sub
  dup
  store 1
  jz done
  jmp loop

done:
  setp16 done_msg
  calln printe
  calln exit

start_msg:
  stringz "SPEED LFSR START"

done_msg:
  stringz "SPEED LFSR DONE"
