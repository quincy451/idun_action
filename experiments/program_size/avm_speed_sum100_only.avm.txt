entry start

start:
  setp16 start_msg
  calln printe
  push16 1000
  store 2

out_loop:
  push16 0
  store 0
  push16 1
  store 1

in_loop:
  load 0
  load 1
  add
  store 0

  load 1
  push16 1
  add
  store 1

  load 1
  push16 101
  lt
  jz out_dec
  jmp in_loop

out_dec:
  load 2
  push16 1
  sub
  dup
  store 2
  jz done
  jmp out_loop

done:
  setp16 done_msg
  calln printe
  calln exit

start_msg:
  stringz "SPEED SUM START"

done_msg:
  stringz "SPEED SUM DONE"
