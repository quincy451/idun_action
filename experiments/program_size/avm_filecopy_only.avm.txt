entry start

start:
  push16 src_name
  calln fopenr
  jz fail

  push16 dst_name
  calln fopenw
  jz fail_close_read

copy_loop:
  calln fread8
  dup
  push16 $ffff
  eq
  jz write_byte
  drop
  calln fcloser
  drop
  calln fclosew
  drop
  calln exit

write_byte:
  calln fwrite8
  jmp copy_loop

fail_close_read:
  calln fcloser
  drop
fail:
  calln exit

src_name:
  stringz "SOURCE.TXT"

dst_name:
  stringz "COPYOUT.TXT"
