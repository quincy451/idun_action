entry start

start:
  ; local 0 = sum, local 1 = i
  push16 0
  store 0
  push16 1
  store 1

sum_loop:
  ; stop when i == 101
  load 1
  push16 101
  eq
  jz done

  ; sum += i
  load 0
  load 1
  add
  store 0

  ; i += 1
  load 1
  push8 1
  add
  store 1
  jmp sum_loop

done:
  load 0
  calln printie
  calln exit
