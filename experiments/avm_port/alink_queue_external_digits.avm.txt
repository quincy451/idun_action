entry start

start:
  push16 in_name
  calln fopenr
  jz exit_now

  push16 out_name
  calln fopenw
  jz close_read_and_exit

  push16 0
  store 4
  store 5
  store 6
  store 7
  store 8
  store 9
  store 10
  store 11
  store 12
  store 13       ; locals 4..13 = seen flags for digits 0..9

scan_loop:
  calln fread8
  dup
  push16 $ffff
  eq
  jz got_char
  drop
  jmp close_both_and_exit

got_char:
  store 0        ; local0 = current opcode char
  load 0
  push8 117      ; 'u'
  eq
  jz handle_external

  load 0
  push8 99       ; c
  eq
  jz skip_pair
  load 0
  push8 115      ; s
  eq
  jz skip_pair
  load 0
  push8 101      ; e
  eq
  jz skip_pair
  load 0
  push8 105      ; i
  eq
  jz skip_pair
  load 0
  push8 106      ; j
  eq
  jz skip_pair
  load 0
  push8 112      ; p
  eq
  jz skip_pair
  load 0
  push8 76       ; L
  eq
  jz skip_pair
  load 0
  push8 83       ; S
  eq
  jz skip_pair
  load 0
  push8 85       ; U
  eq
  jz skip_pair
  load 0
  push8 84       ; T
  eq
  jz skip_pair
  load 0
  push8 90       ; BODY_OP_HELPER_EXT stand-in 'Z'
  eq
  jz skip_pair
  jmp scan_loop

skip_pair:
  calln fread8
  drop
  jmp scan_loop

handle_external:
  calln fread8
  store 1        ; local1 = following digit char
  load 1
  push8 48
  lt
  jz scan_loop
  load 1
  push8 57
  gt
  jz scan_loop

  load 1
  push8 48
  sub
  store 2        ; local2 = digit value 0..9

  load 2
  push8 0
  eq
  jz check1
  load 4
  jmp maybe_emit
check1:
  load 2
  push8 1
  eq
  jz check2
  load 5
  jmp maybe_emit
check2:
  load 2
  push8 2
  eq
  jz check3
  load 6
  jmp maybe_emit
check3:
  load 2
  push8 3
  eq
  jz check4
  load 7
  jmp maybe_emit
check4:
  load 2
  push8 4
  eq
  jz check5
  load 8
  jmp maybe_emit
check5:
  load 2
  push8 5
  eq
  jz check6
  load 9
  jmp maybe_emit
check6:
  load 2
  push8 6
  eq
  jz check7
  load 10
  jmp maybe_emit
check7:
  load 2
  push8 7
  eq
  jz check8
  load 11
  jmp maybe_emit
check8:
  load 2
  push8 8
  eq
  jz check9
  load 12
  jmp maybe_emit
check9:
  load 13

maybe_emit:
  push8 0
  eq
  jz scan_loop

  load 2
  push8 0
  eq
  jz set1
  push8 1
  store 4
  jmp emit_digit
set1:
  load 2
  push8 1
  eq
  jz set2
  push8 1
  store 5
  jmp emit_digit
set2:
  load 2
  push8 2
  eq
  jz set3
  push8 1
  store 6
  jmp emit_digit
set3:
  load 2
  push8 3
  eq
  jz set4
  push8 1
  store 7
  jmp emit_digit
set4:
  load 2
  push8 4
  eq
  jz set5
  push8 1
  store 8
  jmp emit_digit
set5:
  load 2
  push8 5
  eq
  jz set6
  push8 1
  store 9
  jmp emit_digit
set6:
  load 2
  push8 6
  eq
  jz set7
  push8 1
  store 10
  jmp emit_digit
set7:
  load 2
  push8 7
  eq
  jz set8
  push8 1
  store 11
  jmp emit_digit
set8:
  load 2
  push8 8
  eq
  jz set9
  push8 1
  store 12
  jmp emit_digit
set9:
  push8 1
  store 13

emit_digit:
  load 1
  calln fwrite8
  push8 10
  calln fwrite8
  jmp scan_loop

close_both_and_exit:
  calln fcloser
  drop
  calln fclosew
  drop
  calln exit

close_read_and_exit:
  calln fcloser
  drop
exit_now:
  calln exit

in_name:
  stringz "BODY.TXT"
out_name:
  stringz "PENDING.TXT"
