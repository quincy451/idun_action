entry start

start:
  push16 out_name
  calln fopenw
  jz exit_now

  push16 1
  store 0        ; local0 = REU handle placeholder for NUL-terminated name table
  push16 3
  store 1        ; local1 = entry count placeholder
  push16 0
  store 2        ; local2 = current entry index
  push16 0
  store 3        ; local3 = current byte offset in table

entry_loop:
  load 2
  load 1
  eq
  jz close_and_exit

  push8 117      ; 'u'
  calln fwrite8
  push8 32       ; ' '
  calln fwrite8

name_loop:
  load 0
  load 3
  calln reu_peek8
  store 4        ; local4 = current char
  load 4
  push8 0
  eq
  jz emit_char

  push8 10       ; newline after one external name
  calln fwrite8
  load 2
  push8 1
  add
  store 2
  load 3
  push8 1
  add
  store 3
  jmp entry_loop

emit_char:
  load 4
  calln fwrite8
  load 3
  push8 1
  add
  store 3
  jmp name_loop

close_and_exit:
  calln fclosew
  drop
exit_now:
  calln exit

out_name:
  stringz "EXTERN.TXT"
